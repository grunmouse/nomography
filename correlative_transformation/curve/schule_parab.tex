\subsection{Школьная парабола}
$$s: y = ax^2 + bx + c.$$
$$x = f(t) = t,\; y = g(t) = at^2 + bt + c;$$
$$f' = 1; \; g' = 2at + b.$$
$$D = fg' - gf' = 2at^2 + bt - (at^2 + bt + c) = at^2 - c.$$
$$
\left\{
\begin{gathered}
	\bar{x} = -\frac{f'}{D} = -\frac{1}{at^2 - c}, \\
	\bar{y} = \frac{g'}{D} = \frac{2at + b}{at^2 - c};
\end{gathered}
\right.
\Rightarrow
\left\{
\begin{gathered}
	at^2 - c = -\frac{1}{\bar{x}}, \\
	\bar{y} = -\left(2at + b\right)\bar{x};
\end{gathered}
\right.
$$
